The matrix \(P\) is obtained by

nr_animal <- nrow(tbl_ped_p1c)
P = matrix(0, nrow = nr_animal, ncol = nr_animal)
for (i in 1:nr_animal){
  s <- as.integer(tbl_ped_p1c$fid[i])
  d <- as.integer(tbl_ped_p1c$mid[i])
  if (!is.na(s)){
    P[i,s] <- 0.5
  }
  if(!is.na(d)){
    P[i,d] <- 0.5
  }
}
